

/*
  LiquidCrystal Library - Hello World

 Demonstrates the use a 16x2 LCD display.  The LiquidCrystal
 library works with all LCD displays that are compatible with the
 Hitachi HD44780 driver. There are many of them out there, and you
 can usually tell them by the 16-pin interface.

 This sketch prints "Hello World!" to the LCD
 and shows the time.

  The circuit:
 * LCD RS pin to digital pin 12
 * LCD Enable pin to digital pin 11
 * LCD D4 pin to digital pin 5
 * LCD D5 pin to digital pin 4
 * LCD D6 pin to digital pin 3
 * LCD D7 pin to digital pin 2
 * LCD R/W pin to ground
 * LCD VSS pin to ground
 * LCD VCC pin to 5V
 * 10K resistor:
 * ends to +5V and ground
 * wiper to LCD VO pin (pin 3)

 Library originally added 18 Apr 2008

 example added 9 Jul 2009

// include the library code:
#include <LiquidCrystal.h>
#include <Keypad.h>


// initialize the library with the numbers of the interface pins
//LiquidCrystal lcd(rs, enable, d4,d5, d6, d7);
LiquidCrystal lcd(13, 12, 11, 10, 9, 8);
/* Metodo CONSTRUCTOR SOBRECARGADO*/
/* Inicializamos un objeto tipo LiquidCrystal*/


//CONFIGURACION KEYPAD
const byte ROWS = 4; //four rows
const byte COLS = 4; //three columns
char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
byte rowPins[ROWS] = {A4, 2, 3, 4}; //connect to the row pinouts of the keypad
byte colPins[COLS] = {5, 6, 7, A5}; //connect to the column pinouts of the keypad

Keypad keypad = Keypad( makeKeymap(keys), rowPins, colPins, ROWS, COLS );
//CONSTRUCTOR
//FIN CONFIGURQACIOJ KEYPAD

void setup() {
  // set up the LCD's number of columns and rows:
  lcd.begin(16, 2);
  // Print a message to the LCD.
  lcd.print("HOLA PUTITOS!");
  lcd.setCursor(0,1);
   lcd.print("GONORRIENTOS");
   //int numBytes = lcd.print("HOLA PUTITOS!");
  Serial.begin(9600);
}

void loop(){
  int key = keypad.getKey();
  //Serial.println(key);

  if (key != NO_KEY){
    lcd.print(key);
    key -= 48;
    Serial.println(key);
  }
  //delay(100);
}
